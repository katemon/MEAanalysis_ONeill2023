% [contrib, pcvectors] = pcoftraces(TRACES, use_npcs) 
%  calculates principle components for waveforms in a matrix with 
%  one spike per columns of the matrix <TRACES>. <use_npcs> limits 
%  the number of <pcvectors> returned. <contrib> is the contribution
