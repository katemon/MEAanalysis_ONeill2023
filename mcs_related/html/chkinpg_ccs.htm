% [inidx, outidx] = CHKINPG(x,y, varargin) allows you to 
% mark a polygon in graph and find the values within and

% 
% 
% Parameter    values 
% 'plotall'    'on', 'off' (default), plots inside and outside values in different colors. 
% 'fastdraw'   'on', 'off' (default), show the polygon while you draw if set to on 
% 'xyplot'     'on', 'off' (default), plots the values inside the polygon 
%
